737  inline bool Operation::increment(std::vector<int>& x, int position, std::vector<int>& min, std::vector<int>& max) {
738  if (x.empty() or position < 0) return false;
739 
740  bool incremented = true;
741  if (x.at(position) < max.at(position)) {
742  ++x.at(position);
743  } else {
744  incremented = !(position == 0);
745  x.at(position) = min.at(position);
746  --position;
747  if (position >= 0) {
748  incremented = increment(x, position, min, max);
749  }
750  }
751  return incremented;
752  }

786  inline scalar Operation::variance(const std::vector<T>& x, scalar mean) {
787  if (x.empty()) return fl::nan;
788  if (x.size() == 1) return scalar(0.0);
789  scalar result = 0.0;
790  for (std::size_t i = 0; i < x.size(); ++i) {
791  result += (x.at(i) - mean) * (x.at(i) - mean);

793  result /= -1 + x.size();

844  std::string::const_iterator position = str.begin(), next = str.begin();
845  while (next != str.end()) {

851  if (next != str.end()) {
852  position = next + delimiter.size();
853  }
854  }
855  return result;

858  inline std::string Operation::trim(const std::string& text) {
859  if (text.empty()) return text;
860  if (not (std::isspace(text.at(0)) or std::isspace(text.at(text.size() - 1))))
861  return text;
862  std::size_t start = 0, end = text.size() - 1;
863  while (start <= end and std::isspace(text.at(start))) {
864  ++start;
865  }
866  while (end >= start and std::isspace(text.at(end))) {
867  --end;
868  }
869  std::size_t length = end - start + 1;
870  if (length <= 0) return "";
871  return text.substr(start, length);
872  }
