161  inline Linear* Linear::create(const std::string& name,

163  if (not engine) throw Exception("[linear error] cannot create term <" + name + "> "
164  "without a reference to the engine", FL_AT);
165  std::vector<scalar> coefficients;
166  coefficients.push_back((scalar) firstCoefficient);
167 
168  va_list args;

171  coefficients.push_back((scalar) va_arg(args, T));
172  }
173  va_end(args);
174
